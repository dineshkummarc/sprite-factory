Test: jQuery Sprite Factory

Default factory

  • Full IE PNG transparency compatibility
  • Uses two dom elements per sprite
  • Preloading CSS
  • Updatable
var options = {
                src: "./chess.png", 
                width: 60, 
                height: 60, 
                index: [2,6]
              }
var factory = new SpriteFactory(options)
$("#o1").append(factory.generate([1,2]))  
$("#o2").append(factory.generate([0,2]).attr("id", "b"))

Light factory

  • Only 1 dom element per sprite
  • No IE PNG transparency
  • Preloading CSS
  • Updatable
var options = {
                src: "http://..." , 
                width: 60, 
                height: 60, 
                index: [2,6], 
                light: true
              }
var factory2 = new SpriteFactory(options)
$("#o4").append(factory2.generate([1,2]))
$("#o5").append(factory2.generate([1,2]).attr("id", "c"))
